## Install

```bash
npm install --save is-inside-dom-element
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'is-inside-dom-element'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
